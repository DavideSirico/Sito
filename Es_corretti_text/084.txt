#include &ltstdio.h&gt
#include &ltstdlib.h&gt

int main(){
    FILE *fp;
    fp=fopen("C:\\autoexec.bat","r");

    if(fp==NULL){
        printf("Il file non esiste");
        exit(1);
    }

    char buffer;
    while((buffer=fgetc(fp))!=EOF){
        putchar(buffer);
    }
    fclose(fp);
    return 0;
}
