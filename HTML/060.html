void maiusc(char s[], char s2[80]){
    int i,j;
    j=0;
    for(i=0;s[i]!='\0';i++){
        if((s[i]>=65) && (s[i]<=90)){
            s2[j]=s[i];
            j++;
        }
    }
    s2[j]='\0';
}

void sort(char s2[80]){
    int i,j;
    char temp;
    for(i=0;s2[i+1]!='\0';i++){
        for(j=i+1;s2[j]!='\0';j++){
            if(s2[i]>s2[j]){
                temp=s2[i];
                s2[i]=s2[j];
                s2[j]=temp;
            }
        }
    }
}
